1/**
2 * @file ConfigureBT.ino
3 *
4 * HC-05/06 AT Command Center
5 * Version 3.0
6 *
7 * Description: Simple HC05/06 AT configuration program. Requires 2nd UART (Serial1) defined.
8 *
9 * Provides user menu for selecting configuration changes. Automatically identifies
10 * device (HC-05 or HC-06), firmware version, baud and parity settings.
11 * Serial1 automatically configured to match HC-05/06 UART settings. HC-05/06
12 * must be in configuration mode (AT mode) (LED blinking to indicate Not Connected).
13 * Serial monitor settings are 57600 8N1.
14 *
15 * Recent batches of HC-06 appear to have HC-05 firmware (reporting Version 3).
16 * There is no documentation of a Version 3 firmware for HC-06. AT commands differ
17 * for HC-05 firmware, including CR+NL command terminators. Support for this version
18 * has been added beginning with Revision 2 of this software.
19 *
20 * AT response delays:
21 * Around 10~25ms for Version 3.x (newline terminated) - max observed 35ms
22 * Around 500ms for Version 1.x (timeout terminated) - max observed 525ms
23 * Serial writes are asynchronous, so delays must also consider write time
24 *
25 * HC06 connections (for 5V boards - resistors not needed for 3V3):
26 *
27 * TXD <----------------> [Serial 1 RX]
28 * RXD <----+---R_220---> [Serial 1 TX]
29 * |
30 * |
31 * R_330
32 * |
33 * |
34 * Vss
35 *
36 * HC05 connections: same as above, but also include (for AT mode selection)
37 *
38 * CMD <----+---R_220---> [pin 10]
39 * |
40 * |
41 * R_330
42 * |
43 * |
44 * Vss
45 *
46 * Pin connections:
47 * board Mega MKR Uno WiFi Zero Due MSP432
48 * -------------------+-------------------------------------------------------
49 * [Serial 1 RX] | 19 13 0 0 19 3
50 * [Serial 1 TX] | 18 14 1 1 18 4
51 *
52 *
53 * Created on: 18-Oct, 2021
54 * Author: miller4@rose-hulman.edu

57 */

78 while (Serial1.available() > 0) {
79 Serial.print("[HC06]: ");
80 Serial.println(Serial1.readString());
81 Serial.println();
82 }
83
84 // check for user selection of menu option
85 if (Serial.available() > 0) {
86 command = Serial.readString();
87 switch(command.toInt()) {
88 case 1:

91 case 2:
92 setName();
93 break;
94 case 3:

96 break;
97 case 4:
98 setParity();
99 break;
100 case 5:
101 setLocalBaud();
102 break;
103 case 6:
104 setLocalParity();
105 break;
106 case 7:
107 getVersion();
108 break;
109 case 8:
110 scanDevice();
111 break;
112 default:
113 Serial.println("Invalid entry");
114 break;
115 }
116 Serial.println();
117
118 printMenu();
119 }
120
121 delay(SHORT_DELAY);
122}

154/**
155 * responseDelay
156 *
157 * @brief Delays for period necessary to allow completion of HC-xx response to command.
158 *
159 * @param characters count of characters in AT command
160 * @param firmware firmware version identifier for HC-xx
161 * @param command index of AT command (as defined in HC06commands)
162 */
163void responseDelay(unsigned long characters, int firmware, HC06commands command) {
164 if ((baudRate < 0) || (baudRate >= BAUD_LIST_CNT)) return;
165 unsigned long writeMS = (characters + responseChars[command]) * BITS_PER_CHAR * 1000
166 / baudRateList[baudRate];
167 delay(writeMS + responseMS[firmware]);
168}

226/**
227 * scanDevice
228 *
229 * Automated scan of Bluetooth module to determine configuration of UART.
230 * Will identify version of firmware, baud rate, and parity setting, and set
231 * Serial1 to match HC-xx UART settings.
232 *
233 * @returns true if UART configuration successfully identified
234 */
235bool scanDevice() {
236 String comBuffer;
237 firmVersion = FIRM_UNKNOWN;
238 hc06Version = "";
239
240 Serial1.end();
241 delay(CONFIG_DELAY);
242 Serial.print("\nSearching for firmware and version of HC06");
243 // Scan through possible UART configurations for each firmware version.
244 // Use AT command to test for OK response.
245 for (int firmware = FIRM_VERSION3; firmware > FIRM_UNKNOWN; firmware--) {
246 for (parity = NOPARITY; parity < PARITY_LIST_CNT; parity++) {
247 // firmware version 3.x does not support baud rate below 4800
248 if (firmware == FIRM_VERSION3) {
249 baudRate = VERS3_MIN_BAUD;
250 } else {
251 baudRate = 0;
252 }
253 for ( ; baudRate < BAUD_LIST_CNT; baudRate++) {
254 // Test for Version x.x firmware AT echo
255 command = atCommands[ECHO][firmware]; // AT command
256#ifdef DEBUG
257 // debugging instructions to verify characters sent to UART
258 Serial.print("\nCommand length: ");
259 Serial.println(command.length());
260 for (unsigned int i = 0; i < command.length(); i++) {
261 Serial.print("\t");
262 Serial.print(command.charAt(i), HEX);
263 }
264 Serial.println();
265#endif
266 Serial.print(" .");
267 // set to new baud rate and parity setting and test connection
268 Serial1.begin(baudRateList[baudRate], parityList[parity]);
269 delay(CONFIG_DELAY);
270 clearInputStream(firmware);
271 Serial1.print(command);
272 Serial1.flush();
273 responseDelay(command.length(), firmware, ECHO);
274 if (Serial1.available() > 0) {
275 comBuffer = Serial1.readString();
276#ifdef DEBUG
277 Serial.println();
278 Serial.println(comBuffer);
279 for (unsigned int i = 0; i < comBuffer.length(); i++) {
280 Serial.print("\t");
281 Serial.print(comBuffer.charAt(i), HEX);
282 }
283 Serial.println();
284#endif
285 // if OK response received, UART configuration found
286 if (comBuffer.startsWith(STATUS_OK)) {
287 firmVersion = firmware;
288 // firmware version 3.x does not support baud rate below 4800
289 if (firmware == FIRM_VERSION3) {
290 // TODO call getRole then setRole
291 if (getRole() == ROLE_SLAVE) {
292 if (setRole(ROLE_SLAVE)) {
293 deviceModel = MODEL_HC05;
294 } else {
295 deviceModel = MODEL_HC06;
296 }
297 }
298 }
299 break;
300 }
301 }
302 while (Serial1.available() > 0) {
303 // wait until input stream is clear
304 Serial1.read();
305 }
306 // end Test for Version x.x firmware
307 Serial1.end();
308 delay(CONFIG_DELAY);
309 } // end baud rate loop
310 if (VERSION_KNOWN) break;
311 } // end parity loop
312 if (VERSION_KNOWN) break;
313 } // end firmware loop
314 Serial.println();
315
316 // If configuration successfully determined, update firmware version string
317 if (VERSION_KNOWN) {
318 command = atCommands[HCVERSION][firmVersion];
319 Serial1.print(command);
320 Serial1.flush();
321 responseDelay(command.length(), firmVersion, HCVERSION);
322 if (Serial1.available() > 0) {
323 hc06Version = Serial1.readString();
324 }
325 }
326
327 return (VERSION_KNOWN);
328}

366/**
367 * getRole
368 *
369 * Send AT command to request current BT role for HC-05 device.
370 *
371 * @returns ROLE_SLAVE, ROLE_MASTER, or ROLE_SLAVE_LOOP. HC-06 device will
372 * return ROLE_UNKNOWN.
373 */
374int getRole() {
375 String comBuffer = "";
376 int role;
377 command = String(ROLE_REQ);
378 clearInputStream(firmVersion);
379 Serial1.print(command);
380 Serial1.flush();
381 responseDelay(command.length(), firmVersion, ECHO);
382 if (Serial1.available() > 0) {
383 comBuffer = Serial1.readString();
384 Serial.println("\nRequesting device role.");
385 Serial.print("[HC06]: ");
386 Serial.println(comBuffer);
387#ifdef DEBUG
388 for (unsigned int i = 0; i < comBuffer.length(); i++)
389 {
390 Serial.print("\t");
391 Serial.print(comBuffer.charAt(i), HEX);
392 }
393 Serial.println();
394#endif
395 }
396 role = comBuffer.indexOf(':');
397 if (role < 0) {
398 deviceRole = ROLE_UNKNOWN;
399 } else {
400 switch (comBuffer.charAt(role)) {
401 case '0': deviceRole = ROLE_SLAVE;
402 break;
403 case '1': deviceRole = ROLE_MASTER;
404 break;
405 case '2': deviceRole = ROLE_SLAVE_LOOP;
406 break;
407 default: deviceRole = ROLE_UNKNOWN;
408 }
409 }
410 if (deviceRole == ROLE_UNKNOWN) {
411 Serial.println("Role response not identified.");
412 } else {
413 Serial.println("Device role is: " + roleString[deviceRole]);
414 }
415 return deviceRole;
416}

726/**
727 * setPin
728 *
729 * Configure Bluetooth pin of HC-xx device.
730 * Sends AT command to configure BT pin/passkey of HC-xx UART device.
731 * For firmware version 1.x, 4-digit code is accepted. For firmware version
732 * 3.x, up to 16 alphanumeric character passkey is accepted according to
733 * documentation. This is artificially limited to 14 characters to ensure no
734 * conflict with adding quotation characters.
735 */
736void setPin() {
737 String pin;
738 if (firmVersion == FIRM_VERSION3) {
739 Serial.println("Enter new BT passkey (14 characters max): ");
740 } else {
741 Serial.println("Enter new pin number (4 digits): ");
742 }
743
744 while (Serial.available() < 1);
745 pin = Serial.readString();
746 pin.trim();
747 if (firmVersion == FIRM_VERSION3) {
748 // version 3.x FW appears to require quotes around passkey,
749 // though this isn't indicated in documentation
750 // https://forum.arduino.cc/t/password-hc-05/481294
751 pin = String("\"") + pin.substring(0, 14) + String("\"");
752 } else if (pin.length() == 4) {
753 // for firware version 1.x, verify 4 numeric characters received
754 for (unsigned int i = 0; i < 4; i++) {
755 if (!isDigit(pin.charAt(i))) {
756 Serial.println("\nInvalid entry (not 4-digit integer)");
757#ifdef DEBUG
758 Serial.print("\tCharacters: ");
759 for (i = 0; i < pin.length(); i++) {
760 Serial.print(pin[i], HEX);
761 Serial.print(" ");
762 }
763 Serial.println();
764#endif
765 delay(MENU_DELAY);
766 return;
767 }
768 }
769 } else {
770 Serial.println("\nInvalid entry (not 4-digit integer)");
771#ifdef DEBUG
772 Serial.print("\tCharacters: ");
